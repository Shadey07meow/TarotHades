    javac -d bin src/*.java
    java -cp bin  MainGameDriver

    cd Memoriam | javac -d bin src/*.java | java -cp bin  MainGameDriver
    cd Memoriam & javac -d bin src/*.java & java -cp bin  MainGameDriver
